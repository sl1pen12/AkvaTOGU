function generateDates() {
  const dates: Array<{ date: string; day: string; fullDate: string }> = [];
  const today = new Date();
  const days = ['ВС', 'ПН', 'ВТ', 'СР', 'ЧТ', 'ПТ', 'СБ'];
  const months = ['янв', 'фев', 'мар', 'апр', 'май', 'июн', 'июл', 'авг', 'сен', 'окт', 'ноя', 'дек'];
  for (let i = 0; i < 7; i++) {
    const date = new Date(today);
    date.setDate(today.getDate() + i);
    dates.push({
      date: date.getDate() + ' ' + months[date.getMonth()],
      day: days[date.getDay()],
      fullDate: date.toISOString().split('T')[0],
    });
  }
  return dates;
}

function ruDate(dateISO: string) {
  return new Date(dateISO).toLocaleDateString('ru-RU', { day: 'numeric', month: 'long' });
}

export default function SchedulePage({
  userId,
  selectedTariffId,
  onNavigate,
  updateNotificationsCount,
}: SchedulePageProps) {
  const dates = useMemo(() => generateDates(), []);

  const [selectedDate, setSelectedDate] = useState(dates[0].fullDate);
  const [activeTariffId, setActiveTariffId] = useState<TariffId | null>(selectedTariffId || null);
  const [showAll, setShowAll] = useState(!selectedTariffId);

  // Booking modal
  const [showBookingModal, setShowBookingModal] = useState(false);
  const [selectedSession, setSelectedSession] = useState<Session | null>(null);

  // category selection for “purchase now”
  const [selectedCategoryIndex, setSelectedCategoryIndex] = useState(0);

  const { toast } = useToast();

  const [balance, setBalance] = useState<number>(0);

  useEffect(() => {
    if (!userId && userId !== 0) return;
    try {
      const st = ensureUserPoolState(userId as any);
      setBalance(st.balance);
    } catch {
      setBalance(0);
    }
  }, [userId]);

  useEffect(() => {
    if (selectedTariffId) {
      setActiveTariffId(selectedTariffId);
      setShowAll(false);
    }
  }, [selectedTariffId]);

  useEffect(() => {
    loadSessions();
    if (userId) loadPurchases();
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [selectedDate, userId]);

  const loadSessions = async () => {
    setLoading(true);
    try {
      const result = await bookingsAPI.getSessions(selectedDate);
      if (Array.isArray(result) && result.length > 0) setSessions(result);
      else setSessions(TEST_SESSIONS);
    } catch {
      setSessions(TEST_SESSIONS);
    } finally {
      setLoading(false);
    }
  };

  const loadPurchases = async () => {
    if (!userId) return;
    try {
      const userPurchases = await bookingsAPI.getPurchases(userId);
      setPurchases(Array.isArray(userPurchases) ? userPurchases : []);
    } catch {
      setPurchases([]);
    }
  };

  const filteredSessions = useMemo(() => {
    return sessions.filter((s) => {
      if (showAll || !activeTariffId) return true;
      const spec = (s.specialization || '').toLowerCase();
      const tariff = getTariffById(activeTariffId);
      if (!tariff) return true;
      const tariffSpec = tariff.specialization.toLowerCase();
      return spec === tariffSpec || spec.includes(tariffSpec);
    });
  }, [sessions, showAll, activeTariffId]);

  const getActivePurchasesForTariff = () => {
    if (!activeTariffId) return [];
    return purchases.filter((p) => p.tariffId === activeTariffId && p.usedSessions < p.sessionsCount);
  };

  const getRemainingForTariff = (tariffId: number | null | undefined) => {
    if (!tariffId) return 0;
    const list = purchases.filter((p) => p.tariffId === tariffId && p.usedSessions < p.sessionsCount);
    return list.reduce((sum, p) => sum + (p.sessionsCount - p.usedSessions), 0);
  };

  const activeCategories = useMemo(() => {
    if (!activeTariffId) return [];
    return getCategoriesByTariffId(activeTariffId);
  }, [activeTariffId]);

  const hasActivePurchaseForSelectedTariff = useMemo(() => {
    return getActivePurchasesForTariff().length > 0;
  }, [purchases, activeTariffId]);

  const handleBookClick = (session: Session) => {
    setSelectedSession(session);
    setSelectedCategoryIndex(0);
    setShowBookingModal(true);
  };

  const handleBooking = async () => {
    if (!userId || !selectedSession) return;

    // Determine tariff we will use for category selection.
    // If activeTariffId is not set (showAll), force user to choose a tariff.
    if (!activeTariffId) {
      toast({
        title: 'Выберите тариф',
        description: 'Для записи выберите тариф на странице «Тарифы».',
        variant: 'destructive',
      });
      return;

    const tariffName = getTariffNameById(tariffId);

    if (!selectedCategory) {
      toast({
        title: 'Ошибка',
        description: 'Неверно выбранное количество занятий',
        variant: 'destructive',
      });
      return;
    }

    // Иначе — имитируем покупку разового количества (увеличивать balance НЕ будем), но запись оформляем.
    const activePurchase = getActivePurchasesForTariff()[0] || null;
    const remainingByPurchase = activePurchase ? activePurchase.sessionsCount - activePurchase.usedSessions : 0;
    const canUseRemaining = Boolean(activePurchase && remainingByPurchase > 0);

    try {
      const bookedResult = await bookingsAPI.bookSession(userId, selectedSession.id);

      const bookedOk = bookedResult.success === true || (bookedResult && !bookedResult.error);
      if (!bookedOk) {
        toast({
          title: 'Ошибка',
          description: bookedResult.error || 'Не удалось записаться',
          variant: 'destructive',
        });
        return;
      }

      // If we have active paid sessions for this tariff: use purchase session and decrement balance.
      if (activePurchase && hasPaidRemainingForService) {
        await bookingsAPI.usePurchaseSession(userId, activePurchase.id);
        const nextBalance = decrementBalance(userId, 1);
        setBalance(nextBalance);

        await addBookingToHistory(userId, {
          sessionId: selectedSession.id,
          sessionDate: selectedSession.session_date.split('T')[0],
          startTime: selectedSession.start_time.slice(0, 5),
          endTime: selectedSession.end_time?.slice(0, 5),
          tariffName,
          categoryName: undefined,
          specialization: selectedSession.specialization || 'Занятие',
          trainerName: selectedSession.trainer_name,
          status: 'предстоит',
          paidOnSpot: false,
        });

        await bookingsAPI.addNotification(userId, {
          type: 'booking',
          title: 'Запись',
          message: `Запись подтверждена. Осталось занятий: ${Math.max(0, nextBalance)}.` ,

          sessionDate: selectedSession.session_date,

          trainerName: selectedSession.trainer_name,
          tariffName,
        });

        if (updateNotificationsCount) updateNotificationsCount();

        toast({ title: 'Успешно!', description: 'Запись подтверждена.' });
        setShowBookingModal(false);
        return;
      }

      // Otherwise: on-spot payment imitation.
      // Requirement:
      // - If user has balance==0 we keep balance==0 and do not increase.
      // - We allow booking to pass, and mark history as paidOnSpot.
      // - If user has balance>0 but no active purchase for this tariff, we still allow on-spot purchase simulation in UI,

      await addBookingToHistory(userId, {
        sessionId: selectedSession.id,
        sessionDate: selectedSession.session_date.split('T')[0],
        startTime: selectedSession.start_time.slice(0, 5),

      });

        description: 'Занятие забронировано. Оплата — на месте (имитация).',

    } catch (e: any) {

        <div className="flex items-center justify-between flex-wrap gap-2">
          <h2 className="font-semibold text-gray-900 flex items-center gap-2">
            <Icon name="Clock" size={18} className="text-primary" />
            Ваш активный абонемент
          </h2>

          {!showAll && activeTariffId && (
            <div className="flex items-center gap-2">
              <Badge variant="default" className="bg-[#9c1d3b] text-base">
                {getTariffNameById(activeTariffId)}
              </Badge>
              <Button
                variant="outline"
                size="sm"
                onClick={() => {
                  setShowAll(true);
                  setActiveTariffId(null);
                  localStorage.removeItem('selected_tariff_id');
                }}
              >
                Сброс
              </Button>
            </div>
          )}
        </div>

        {/* Active purchase card */}
        {!showAll && activeTariffId && (
          <Card className="p-4 bg-gradient-to-r from-blue-50 to-indigo-50 border-2 border-blue-200">
            <div className="flex items-center gap-2 mb-3">
              <div className="p-2 bg-blue-500 rounded-lg">
                <Icon name="CreditCard" size={20} className="text-white" />
              </div>
              <div>
                <p className="font-bold text-gray-900">Абонемент активен</p>
                <p className="text-xs text-gray-600">{getTariffNameById(activeTariffId)}</p>
              </div>
            </div>

            {remainingForActiveTariff > 0 ? (
              <div className="bg-white rounded-xl p-4 shadow-sm">
                <div className="flex items-center justify-between">
                  <span className="font-bold text-gray-900">У вас осталось:</span>
                  <Badge variant="default" className="bg-green-600 text-lg px-4 py-2 h-auto">
                    {remainingForActiveTariff}
                  </Badge>
                </div>
                <p className="text-xs text-gray-600 mt-2">
                  Счётчик показывает только оплаченные занятия по выбранному тарифу.
                </p>
              </div>
            ) : (
              <div className="flex items-center gap-2 text-blue-700">
                <Icon name="Info" size={18} />
                <span className="text-sm">Оплата/покупка нужна для записи по этому виду</span>
              </div>
            )}
          </Card>
        )}

        {/* Sessions list */}
        {loading ? (
          <Card className="p-8 text-center">
            <p className="text-gray-600">Загрузка...</p>
          </Card>
        ) : filteredSessions.length === 0 ? (
          <Card className="p-8 text-center">
            <Icon name="CalendarX" size={40} className="mx-auto text-gray-300 mb-2" />
            <p className="text-gray-600">Сеансов нет</p>
          </Card>
        ) : (
          <div className="space-y-2">
            {filteredSessions.map((session) => (
              <Card
                key={session.id}
                className={
                  'p-3 transition-all ' +
                  (session.available_slots === 0
                    ? 'opacity-60 bg-gray-50'
                    : 'hover:shadow-md hover:border-[#9c1d3b]')
                }

                // If there is paid remaining for selected tariff => show remaining and simple booking.

                          <p className="text-xs text-green-700">Подтверждение оплаты не требуется</p>
                        </div>
                      </div>
                    </div>
                  );
                }

                // Otherwise => allow purchase of category amount.
                return (
                  <div>
                    <div className="flex items-center gap-2 mb-3">
                      <div className="p-2 bg-[#9c1d3b]/10 rounded-lg">
                        <Icon name="CreditCard" size={18} className="text-[#9c1d3b]" />
                      </div>
                      <div>
                        <p className="font-bold text-[#9c1d3b]">Осталось занятий: 0</p>
                        <p className="text-xs text-gray-600">Требуется оплата занятия</p>
                      </div>
                    </div>

                    <p className="text-sm text-gray-600 mb-2 flex items-center gap-1">
                      <Icon name="Info" size={14} />
                      Выберите, сколько занятий купить (имитация):
                    </p>

                    <div className="space-y-2 max-h-48 overflow-y-auto">
                      {activeCategories.map((cat, idx) => (
                        <label
                          key={idx}
                          className={
                            'w-full p-3 rounded-lg border-2 transition-all cursor-pointer ' +
                            (selectedCategoryIndex === idx
                              ? 'border-[#9c1d3b] bg-[#9c1d3b]/10'
                              : 'border-gray-200 hover:border-[#9c1d3b]/50')
                          }
                        >
                          <div className="flex items-center gap-3">
                            <input
                              type="radio"
                              name="abonent"
                              checked={selectedCategoryIndex === idx}
                              onChange={() => setSelectedCategoryIndex(idx)}
                              className="w-4 h-4 text-[#9c1d3b]"
                            />
                            <div className="flex-1">
                              <span className="font-medium text-gray-900">{cat.name}</span>
                              <p className="text-xs text-gray-500 mt-0.5">{cat.sessions} занятие(й)</p>
                            </div>
                            <Badge variant={selectedCategoryIndex === idx ? 'default' : 'secondary'} className="text-base shrink-0">
                              {cat.price}
                            </Badge>
                          </div>
                        </label>
                      ))}
                    </div>

                    <p className="text-xs text-blue-600 mt-2 flex items-center gap-1">
                      <Icon name="Info" size={12} />
                      При подтверждении запись будет оформлена, баланс останется 0 (имитация оплаты на месте).
                    </p>
                  </div>
                );
              })()}
            </div>

            <div className="flex gap-2">

}
